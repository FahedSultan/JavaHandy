    while(a<b){
    for(int x=0;x<20;x++) {
    	p=isPrime(a,b);

    }
    }

public  static int isPrime(int a, int b){
	int x = -999;
    while (a<b){
	    int n;
	    n=a+1;
	    n++;
		for ( int i=2; i<b ;i++ ) {
			x = (n%i);
           if(i == n)
        	   break;
	    }

	 }
    return x;
}
